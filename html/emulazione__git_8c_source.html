21void Inizializza_GIT(GIT *var_git)
22{
23 var_git->duty_A = 50; /* Valore di default per duty cycle = 50% */
24 var_git->duty_B = 50;
25 var_git->fase = 90; /* Valore di default per sfasamento = 90 gradi */
26
27 var_git->ris = 128; /* Valore di default per
28 risoluzione encoder = 128 ppr */
29
30 var_git->diam = 1; /* Valore di default per
31 diametro ruota = 1 metro */

34 lunghezza
35 passo*/
36

45 passo */
46
47 var_git->n_passi = 0; /* Reset passi compiuti */

52/* Qua si fa il grosso del lavoro, ricalcolo la velocità della ruota, e lo
53spazio in cui si trovano i canali */
54void Aggiorna_spazio_e_vel(GIT *var_git)
55{
56 long double pos_minore; /* Contiene la posizione minore tra quella
57 del canale A e B, serve per la correzione
58 dello spazio */
59
60 long double pos_maggiore; /* Contiene la posizione maggiore tra quella
61 del canale A e B, serve per la correzione
62 dello spazio */
63
64 /* Integrazione dell'accelerazione */
65 var_git->vel = var_git->acc * T_SAMPLE + var_git->vel;
66
67 /* Saturo la velocità se va oltre la soglia di 700 km/h*/
68 if (var_git->vel > VELOCITA_MAX)
69 {
70 var_git->vel = VELOCITA_MAX;
71 }
72 else if (var_git->vel < -VELOCITA_MAX)
73 {
74 var_git->vel = -VELOCITA_MAX;
75 }
76
77 /* Integrazione della velocità, assegno lo spazio ad A
78 (che è una scelta arbitraria) */
79 var_git->pos_A = var_git->vel * T_SAMPLE + var_git->pos_A;
80
81 /* Estrapolo il fattore di sfasamento, cioè converto i gradi nella
82 quantità di spazio da cui il canale B si discosta dal canale A */
83 float k_fase = ((float) - (var_git->fase) / 360);
84 var_git->pos_B = var_git->pos_A + var_git->passo * k_fase;
85
86 /* Indico chi è il sensore con la posizione maggiore e quale con la
87 minore, mi serve per fare il controllo sulla correzione di spazio */
88 if (k_fase <= 0)
89 {
90 /* Posizione del B è minore dell'A */
91 pos_minore = var_git->pos_B;
92 pos_maggiore = var_git->pos_A;
93 }
94 else
95 {
96 /* Posizione dell'A è minore del B */
97 pos_minore = var_git->pos_A;
98 pos_maggiore = var_git->pos_B;
99 }
100
101 /* Satura lo spazio se uno dei due sensori va oltre la
102 soglia [-2*passi, 2*passi] */
103 if (pos_minore <= -2 * var_git->passo)
104 {
105 /* il minore ha sforato, esso ritorna a 0 mentre il maggiore
106 va ad un valore maggiore di 0 (dipende dalla fase) */
107 var_git->pos_A = var_git->pos_A + (2 * var_git->passo);
108 var_git->pos_B = var_git->pos_B + (2 * var_git->passo);
109 }
110 else if (pos_maggiore >= 2 * var_git->passo)
111 {
112 /* il maggiore ha sforato, esso ritorna a 0 mentre il minore
113 va ad un valore minore di 0 (dipende dalla fase) */
114 var_git->pos_A = var_git->pos_A - (2 * var_git->passo);
115 var_git->pos_B = var_git->pos_B - (2 * var_git->passo);
116 }
117 else
118 {
119 /* Richiesto dal MISRA, non succede nulla */
120 }
121
122 return;
123}

125/* Con questa funzione scrivo su GPIO lo stato dei canali
126 con annesso report */
127uint8_t Emula_GIT_wGPIO(GIT *var_git, XGpio GPIO_CH_A, XGpio GPIO_CH_B)
128{
129 /* Stato del canale, equivalente al valore logico di GPIO corrispondente */
130 uint8_t statoA;
131 uint8_t statoB;
132
133 /* Porto i duty cycle da percentuale intera a decimale*/
134 float k_dutyA = ((float) var_git->duty_A) * 0.01;
135 float k_dutyB = ((float) var_git->duty_B) * 0.01;
136
137 /* Genero le soglie su cui fare il confronto, per stimare in che
138 posizione è il sensore */
139 long double soglia_max_def = 2 * var_git->passo; /* 2passi */
140 long double soglia_min_def = - soglia_max_def; /* -2passi */
141 /* Duty% * 2passi */
142 long double soglia_pos_A = k_dutyA * soglia_max_def;
143 long double soglia_pos_B = k_dutyB * soglia_max_def;
144 /* (1-Duty%) * 2passi */
145 long double soglia_neg_A = (1 - k_dutyA) * soglia_min_def;
146 long double soglia_neg_B = (1 - k_dutyB) * soglia_min_def;
147
148 /* Generazione segnale per canale A da GPIO,
149 se il treno va a velocità positiva allora lo spazio sarà positivo,
150 quindi verranno usati solo gli ultimi 2 if dei 4 totali, che fanno
151 il check per posizioni positive. Ogni paio di if dice se la GPIO è
152 alto o basso, la soglia che decide questo dipende dal duty cycle.
153 Esempio: il range dello spazio è 0 -> 2 passi, se duty = 50%
154 la soglia di transizione alto/basso sarà 1 passo. */
155 if (var_git->pos_A >= soglia_min_def && var_git->pos_A < soglia_neg_A)
156 {
157 XGpio_DiscreteWrite(&GPIO_CH_A, 1, 0x01);
158 statoA = 1;
159 }
160 else if (var_git->pos_A >= soglia_neg_A && var_git->pos_A < 0)
161 {
162 XGpio_DiscreteWrite(&GPIO_CH_A, 1, 0x00);
163 statoA = 0;
164 }
165 else if (var_git->pos_A >= 0 && var_git->pos_A < soglia_pos_A)
166 {
167 XGpio_DiscreteWrite(&GPIO_CH_A, 1, 0x01);
168 statoA = 1;
169 }
170 else if (var_git->pos_A >= soglia_pos_A && var_git->pos_A < soglia_max_def)
171 {
172 XGpio_DiscreteWrite(&GPIO_CH_A, 1, 0x00);
173 statoA = 0;
174 }
175
176 /* Generazione segnale per canale B da GPIO, valgono gli stessi
177 ragionamenti del canale A*/
178 if (var_git->pos_B >= soglia_min_def && var_git->pos_B < soglia_neg_B)
179 {
180 XGpio_DiscreteWrite(&GPIO_CH_B, 1, 0x01);
181 statoB = 1;
182 }
183 else if (var_git->pos_B >= soglia_neg_B && var_git->pos_B < 0)
184 {
185 XGpio_DiscreteWrite(&GPIO_CH_B, 1, 0x00);
186 statoB = 0;
187 }
188 else if (var_git->pos_B >= 0 && var_git->pos_B < soglia_pos_B)
189 {
190 XGpio_DiscreteWrite(&GPIO_CH_B, 1, 0x01);
191 statoB = 1;
192 }
193 else if (var_git->pos_B >= soglia_pos_B && var_git->pos_B < soglia_max_def)
194 {
195 XGpio_DiscreteWrite(&GPIO_CH_B, 1, 0x00);
196 statoB = 0;
197 }
198
199 /* Stato della macchina a stati equivalente del git, le variabili di stato
200 sono -> statoA & statoB, quindi:
201 00 -> stato 0
202 01 -> stato 1
203 11 -> stato 2
204 10 -> stato 3 */
205 uint8_t stato_git;
206
207 if (statoA == 0)
208 {
209 if (statoB == 0)
210 {
211 /* 00 -> stato 0 */
212 stato_git = 0;
213 }
214 else
215 {
216 /* 01 -> stato 1 */
217 stato_git = 1;
218 }
219 }
220 else
221 {
222 if (statoB == 0)
223 {
224 /* 10 -> stato 3 */
225 stato_git = 3;
226 }
227 else
228 {
229 /* 11 -> stato 2 */
230 stato_git = 2;
231 }
232 }
233
234 return stato_git;
235
236}
