85/* Inizializza l'Interrupt System */
86int Setup_Interrupt_System(XScuGic *IstanzaGIC)
87{
88 int Status;
89
90 XScuGic_Config *IntcConfig; /* Istanza dell'interrupt controller */
91
92 Xil_ExceptionInit();
93
94 /* Inizializza i driver dell'interrupt controller */
95 IntcConfig = XScuGic_LookupConfig(INTC_DEVICE_ID);
96 if (NULL == IntcConfig) {
97 return XST_FAILURE;
98 }
99
100 Status = XScuGic_CfgInitialize(IstanzaGIC, IntcConfig,
101 IntcConfig->CpuBaseAddress);
102
103 /* Collecga l'handler dell'interrupt alla logica di gestione di
104 * di interrupt hw del processore */
105 Xil_ExceptionRegisterHandler(XIL_EXCEPTION_ID_INT,
106 (Xil_ExceptionHandler)XScuGic_InterruptHandler,
107 IstanzaGIC);
108
109 /* Abilita l'interrupt del processore */
110 Xil_ExceptionEnableMask(XIL_EXCEPTION_IRQ);
111
112 if (Status != XST_SUCCESS) {
113 return XST_FAILURE;
114 }
115
116 return XST_SUCCESS;
117}

181void Loop_interrupt(void *CallBack_Timer)
182{
183 uint8_t stato_GIT1;
184 uint8_t stato_GIT2;
185
186 /* Rimuovi flag di interrupt */
187 XScuTimer_ClearInterruptStatus(&IstanzaTimer);
188
189 counter_50ms++;
190 /* Frequenza dell'interrupt, f_i = 250 kHz*/
191 /* Frequenza di lettura della fifo, f_f= 2o Hz */
192 /* Devo contare (f_i/f_f) = 12500 interrupt */
193 if (counter_50ms == 12500){
194 /* Leggo la fifo dei telegrammi ogni 50 ms*/
195 Aggiornamento_variabili(&fifo_telegrammi, &git1, &git2);
196
197 /* Se una Minized non si è collegata allora mando
198 un messaggio identificativo */
199 if (0 == fifo_uart.flag_collegamento) xil_printf("WAM\n");
200 counter_50ms = 0;
201 }
202
203 /* Aggiorno le variabili di stato interne di un GIT (es. velocità) */
204 Aggiorna_cinematica(&git1);
205 Aggiorna_cinematica(&git2);
206
207 /* Scrivo su GPIO la forma d'onda del GIT */
208 stato_GIT1 = Emula_GIT_wGPIO(&git1, GPIO_GIT_1_CANALE_A,
209 GPIO_GIT_1_CANALE_B);
210 stato_GIT2 = Emula_GIT_wGPIO(&git2, GPIO_GIT_2_CANALE_A,
211 GPIO_GIT_2_CANALE_B);
212
213 /* Controllo lo stato della FSM equivalente di un GIT, se lo stato
214 presente e lo stato passato sono diversi allora significa che il GIT
215 ha fatto un passo */
216 if(stato_GIT1 != stato_GIT1_prec)
217 {
218 git1.n_passi++;
219 }
220
221 if(stato_GIT2 != stato_GIT2_prec)
222 {
223 git2.n_passi++;
224 }
225
226 stato_GIT1_prec = stato_GIT1;
227 stato_GIT2_prec = stato_GIT2;
228}
